```go
func TestNOPOut_ExpStatSNConfirmation(t *testing.T) {
    rec := &pducapture.Recorder{}
    tgt, err := testutil.NewMockTarget()
    // ... setup with HandleLogin, HandleLogout, HandleNOPOut, HandleSCSIFunc ...

    sess, err := uiscsi.Dial(ctx, tgt.Addr(),
        uiscsi.WithPDUHook(rec.Hook()),
        uiscsi.WithKeepaliveInterval(30*time.Second), // long, to avoid auto pings
    )
    // Send a SCSI command first to establish CmdSN
    _ = sess.TestUnitReady(ctx, 0)

    // Trigger ExpStatSN confirmation
    err = sess.SendExpStatSNConfirmation()
    if err != nil {
        t.Fatalf("SendExpStatSNConfirmation: %v", err)
    }
    time.Sleep(200 * time.Millisecond)

    // Find the NOP-Out with ITT=0xFFFFFFFF AND TTT=0xFFFFFFFF
    nopouts := rec.Sent(pdu.OpNOPOut)
    var found *pdu.NOPOut
    for _, cap := range nopouts {
        nop := cap.Decoded.(*pdu.NOPOut)
        if nop.InitiatorTaskTag == 0xFFFFFFFF && nop.TargetTransferTag == 0xFFFFFFFF {
            found = nop
            break
        }
    }
    if found == nil {
        t.Fatal("no ExpStatSN confirmation NOP-Out found (ITT=TTT=0xFFFFFFFF)")
    }
    // Assert fields per FFP #15.3:
    if !found.Header.Immediate {
        t.Error("I-bit not set")
    }
    if !found.Header.Final {
        t.Error("F-bit not set")
    }
    // CmdSN should NOT be advanced — verify next command CmdSN is same as before
}
```

If `sendExpStatSNConfirmation` is already public from Plan 01, skip adding the wrapper. If it is unexported, add `SendExpStatSNConfirmation() error` as a public wrapper in `internal/session/keepalive.go` (and then it gets exported through the public `Session` type in the root package).

- At least one LogoutReq was sent

- CmdSN is valid (incremented from last command)

- CmdSN is valid
- ExpStatSN matches expected
- DataSegmentLen = 0
